\tikzstyle{every picture}=[line width=0.05cm, cap=butt, color=black]

% Define some useful tikz macros

% Draw an arc from its center (and not the starting point).
% Adapted from the comment of Tom Bombadil on: https://tex.stackexchange.com/questions/66216
\def\centerarc[#1](#2)(#3:#4:#5) % [draw options] (center) (initial angle:final angle:radius), e.g.: \centerarc[red,thick](0,0)(5:85:1)
    { \draw[#1] ($(#2)+({#5*cos(#3)},{#5*sin(#3)})$) arc (#3:#4:#5); }
